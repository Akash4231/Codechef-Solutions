typedef long long ll;

//-----------------------Modular Arithmetic---------------//

const ll mod = 1e9 + 7;
inline ll add(ll x, ll y) {
    x += y;
    if (x >= mod) return x - mod;
    return x;
}
inline ll sub(ll x, ll y) {
    x -= y;
    if (x < 0) return x + mod;
    return x;
}
inline ll mul(ll x, ll y) {
    return (x * 1ll * y) % mod;
}
inline ll power(ll x, ll y) {
    ll ans = 1;
    while (y) {
        if (y & 1) ans = mul(ans, x);
        x = mul(x, x);
        y >>= 1;
    }
    return ans;
}
inline ll inv(ll x) {
    return power(x, mod - 2);
}

void solve()
{
    int n, k;
    cin >> n >> k;
    string ans = "";
    int num = 25;

    while (num >= 0 && n) {
        int val = (1 << num);
        if (k - val < n - 1) {
            --num;
            continue;
        }
        --n;
        k -= val;
        ans += ((char)('a' + num));
    }

    if (n == 0 && k == 0) cout << ans;
    else cout << "-1";
}
